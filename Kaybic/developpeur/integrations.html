Intégration Personnalisée

Construisez votre propre intégration avec nos APIs REST

Exemple d'Intégration Custom 
curl -X POST https://api.kaybic.africa/v1/payments \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "amount": 10000,
    "currency": "XOF",
    "reference": "PAY_001",
    "customer": {
      "name": "John Doe",
      "phone": "+22670123456"
    },
    "provider": "orange_money",
    "callback_url": "https://votre-site.com/webhook"
  }'
<?php
require_once 'vendor/autoload.php';

use Kaybic\KaybicClient;

$client = new KaybicClient('YOUR_API_KEY');

$payment = $client->payments()->create([
    'amount' => 10000,
    'currency' => 'XOF',
    'reference' => 'PAY_001',
    'customer' => [
        'name' => 'John Doe',
        'phone' => '+22670123456'
    ],
    'provider' => 'orange_money',
    'callback_url' => 'https://votre-site.com/webhook'
]);

echo "Payment URL: " . $payment->payment_url;
?>
const Kaybic = require('@kaybic/node-sdk');

const client = new Kaybic('YOUR_API_KEY');

async function createPayment() {
  try {
    const payment = await client.payments.create({
      amount: 10000,
      currency: 'XOF',
      reference: 'PAY_001',
      customer: {
        name: 'John Doe',
        phone: '+22670123456'
      },
      provider: 'orange_money',
      callback_url: 'https://votre-site.com/webhook'
    });
    
    console.log('Payment URL:', payment.payment_url);
  } catch (error) {
    console.error('Error:', error.message);
  }
}
import kaybic

client = kaybic.Client('YOUR_API_KEY')

payment = client.payments.create(
    amount=10000,
    currency='XOF',
    reference='PAY_001',
    customer={
        'name': 'John Doe',
        'phone': '+22670123456'
    },
    provider='orange_money',
    callback_url='https://votre-site.com/webhook'
)

print(f"Payment URL: {payment.payment_url}")
